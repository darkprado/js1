  1. var a = 1, b = 1, c, d; - объявили переменные a, b, c, d. a и b сразу при объявлении присвоили значение - 1, c и d - undefined
    c = ++a; alert(c); // 2 - к переменной a прибавили единицу (а == 2), и присвоили значение переменной а переменной с; вывели переменную с (2)
    d = b++; alert(d); // 1 - переменной d присвоили значение переменной b(1), затем к b прибавили единицу (b == 2); вывели d(1)
    c = (2+ ++a); alert(c); // 5 - a(2) увеличиваем на единицу(а == 3), теперь переменной c присваиваем значение выражения 2 + 3(с == 5); выводим с(5)
    d = (2+ b++); alert(d); // 4 - переменной d присваиваем значение выражения 2 + b (b == 2), d == 4, после присваивания увеличиваем b на единицу (b == 3); выводим d(4)
    alert(a); // 3 - выводим a(3)
    alert(b); // 3 - выводим b(3)
